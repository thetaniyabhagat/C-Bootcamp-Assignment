3. Write a program to swap values of two int variables

5. Write a program to input a three-digit number and display the sum of the digits.
Ans.  #include<stdio.h>  
             int main()    
             {    
                  int n,sum=0,m;    
                  printf("Enter a number:");    
                  scanf("%d",&n);    
                  while(n>0)    
                   {    
                        m=n%10;    
                        sum=sum+m;    
                        n=n/10;    
                   }    
                 printf("Sum is=%d",sum);    
                 return 0;  
              }                   
OUTPUT:
Enter a number:654
Sum is=15
